### Character interior background
character_location_exterior = {
	texture = {
		reference = "terrain_types/savanna.dds"

		trigger = {
			NOT = {
				exists = liege
			}
			exists = location
			location = {
				terrain = savanna
			}
		}
	}

	texture = {
		reference = "terrain_types/high_boreal.dds"
